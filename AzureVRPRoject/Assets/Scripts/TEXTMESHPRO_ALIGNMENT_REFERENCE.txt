// TextMeshPro TextAlignmentOptions Reference
// ==========================================
//
// In newer versions of TextMeshPro, some TextAlignmentOptions enum values have been renamed
// for better consistency and clarity. This file documents the changes.
//
// DEPRECATED → CURRENT MAPPING:
//
// Horizontal Alignment:
// - TextAlignmentOptions.Left              → TextAlignmentOptions.Left (unchanged)
// - TextAlignmentOptions.Center            → TextAlignmentOptions.Center (unchanged)  
// - TextAlignmentOptions.Right             → TextAlignmentOptions.Right (unchanged)
// - TextAlignmentOptions.Justified         → TextAlignmentOptions.Justified (unchanged)
//
// Vertical Alignment:
// - TextAlignmentOptions.Top               → TextAlignmentOptions.Top (unchanged)
// - TextAlignmentOptions.Middle            → TextAlignmentOptions.Middle (unchanged)
// - TextAlignmentOptions.Bottom            → TextAlignmentOptions.Bottom (unchanged)
// - TextAlignmentOptions.Baseline          → TextAlignmentOptions.Baseline (unchanged)
//
// Combined Alignment (THESE CHANGED):
// - TextAlignmentOptions.TopLeft           → TextAlignmentOptions.TopLeft (unchanged)
// - TextAlignmentOptions.TopCenter         → TextAlignmentOptions.TopCenter (unchanged)
// - TextAlignmentOptions.TopRight          → TextAlignmentOptions.TopRight (unchanged)
// - TextAlignmentOptions.MidlineLeft       → TextAlignmentOptions.MidlineLeft (unchanged)
// - TextAlignmentOptions.MidlineCenter     → TextAlignmentOptions.MidlineCenter (unchanged)
// - TextAlignmentOptions.MidlineRight      → TextAlignmentOptions.MidlineRight (unchanged)
// - TextAlignmentOptions.BottomLeft        → TextAlignmentOptions.BottomLeft (unchanged)
// - TextAlignmentOptions.BottomCenter      → TextAlignmentOptions.BottomCenter (unchanged)
// - TextAlignmentOptions.BottomRight       → TextAlignmentOptions.BottomRight (unchanged)
//
// DEPRECATED VALUES (use the new names):
// - TextAlignmentOptions.CenterLeft        → TextAlignmentOptions.MidlineLeft
// - TextAlignmentOptions.CenterRight       → TextAlignmentOptions.MidlineRight
//
// USAGE EXAMPLES:
//
// For left-aligned text at vertical center:
// textComponent.alignment = TextAlignmentOptions.MidlineLeft;
//
// For right-aligned text at vertical center:
// textComponent.alignment = TextAlignmentOptions.MidlineRight;
//
// For centered text:
// textComponent.alignment = TextAlignmentOptions.Center;
//
// For top-left aligned text:
// textComponent.alignment = TextAlignmentOptions.TopLeft;
//
// FIXED IN THIS PROJECT:
// - VRMenuSystem.cs: CenterLeft → MidlineLeft
// - VRMenuSystem.cs: CenterRight → MidlineRight
